# BROski Bot v3.0 → v4.0 Migration Guide

This guide provides step-by-step instructions for migrating from the prototype v3.0 to the production-ready v4.0 architecture.

---

## 📋 Pre-Migration Checklist

### Data Backup
- [ ] Export current SQLite database: `sqlite3 broski.db .dump > backup.sql`
- [ ] Backup `.env` file
- [ ] Document current bot token and configuration
- [ ] Export user balances and streaks
- [ ] Save active focus sessions

### Environment Preparation
- [ ] Install PostgreSQL 15+
- [ ] Install Redis 7+
- [ ] Install Python 3.11+
- [ ] Install Docker & Docker Compose
- [ ] Create new GitHub branch: `git checkout -b v4-refactor`

## 🔄 Phase 2: Data Migration (Week 2)

### Step 2.1: Extract Data from SQLite
```python
# scripts/extract_sqlite_data.py
import sqlite3
import json
from datetime import datetime

# Connect to old database
conn = sqlite3.connect('broski.db')
cursor = conn.cursor()

# Extract users
cursor.execute("SELECT * FROM users")
users = cursor.fetchall()

# Extract economy data
cursor.execute("SELECT * FROM economy")
economy = cursor.fetchall()

# Extract focus sessions
cursor.execute("SELECT * FROM focus_sessions")
sessions = cursor.fetchall()

# Save to JSON for migration
data = {
    'users': [dict(zip([col[0] for col in cursor.description], row)) for row in users],
    'economy': [dict(zip([col[0] for col in cursor.description], row)) for row in economy],
    'sessions': [dict(zip([col[0] for col in cursor.description], row)) for row in sessions],
}

with open('migration_data.json', 'w') as f:
    json.dump(data, f, indent=2, default=str)

print(f"Extracted {len(users)} users, {len(economy)} economy records, {len(sessions)} sessions")
```

### Step 2.2: Import Data to PostgreSQL
```python
# scripts/import_to_postgres.py
import asyncio
import json
from datetime import datetime

from src.core.database import db
from src.models import User, Economy, FocusSession

async def migrate_data():
    await db.init()
    
    # Load migration data
    with open('migration_data.json', 'r') as f:
        data = json.load(f)
    
    async with db.session() as session:
        # Migrate users
        for user_data in data['users']:
            user = User(
                id=user_data['id'],
                username=user_data['username'],
                discriminator=user_data['discriminator'],
                level=user_data.get('level', 1),
                xp=user_data.get('xp', 0),
                created_at=datetime.fromisoformat(user_data['created_at']),
            )
            session.add(user)
        
        # Migrate economy
        for econ_data in data['economy']:
            economy = Economy(
                user_id=econ_data['user_id'],
                balance=econ_data['balance'],
                lifetime_earned=econ_data.get('lifetime_earned', 0),
                daily_streak=econ_data.get('daily_streak', 0),
                last_daily_claim=datetime.fromisoformat(econ_data['last_daily_claim']) if econ_data.get('last_daily_claim') else None,
            )
            session.add(economy)
        
        # Migrate focus sessions
        for session_data in data['sessions']:
            focus_session = FocusSession(
                id=session_data['id'],
                user_id=session_data['user_id'],
                project_name=session_data['project_name'],
                start_time=datetime.fromisoformat(session_data['start_time']),
                end_time=datetime.fromisoformat(session_data['end_time']) if session_data.get('end_time') else None,
                duration_minutes=session_data.get('duration_minutes'),
                tokens_earned=session_data.get('tokens_earned', 0),
                is_active=session_data.get('is_active', False),
            )
            session.add(focus_session)
        
        await session.commit()
    
    print("Migration complete!")

if __name__ == "__main__":
    asyncio.run(migrate_data())
```

### Step 4.4: Add Rate Limiting
```python
# Add rate limiting decorator
from functools import wraps
import time
from collections import defaultdict

user_cooldowns = defaultdict(dict)

def rate_limit(commands_per_minute=10):
    def decorator(func):
        @wraps(func)
        async def wrapper(self, interaction, *args, **kwargs):
            user_id = interaction.user.id
            now = time.time()
            
            # Clean old timestamps
            if user_id in user_cooldowns:
                user_cooldowns[user_id] = {
                    cmd: ts for cmd, ts in user_cooldowns[user_id].items()
                    if now - ts < 60
                }
            
            # Check rate limit
            if len(user_cooldowns[user_id]) >= commands_per_minute:
                await interaction.response.send_message(
                    "⏰ You're sending commands too fast. Please slow down.",
                    ephemeral=True
                )
                return
            
            # Update cooldown
            user_cooldowns[user_id][func.__name__] = now
            
            return await func(self, interaction, *args, **kwargs)
        return wrapper
    return decorator
```
